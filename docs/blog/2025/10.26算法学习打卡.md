using namespace std;

    vector<int>a(n,0);
    int s,m;
    for(int i=0;i<n-1;++i){
        cin>>s>>m;
        a[s-1]++;
        a[m-1]++;
    }
    if(n==2){
        cout<<1<<" "<<1;

    }

    return 0;
}
