#include <bits/stdc++.h>
using namespace std;
#define ll long long
const ll mod=998244353;
ll fast_pow(ll a,ll b,ll p){
    a %=p;
    ll ans =1;
    while(b){
        if(b&1)//判断b是奇数
        ans=ans*a%p;
        a =a*a%p;
        b >>=1;
        
    }
    return ans;
}


int main() {
    int T;
    cin>>T;
    ll n,con;
